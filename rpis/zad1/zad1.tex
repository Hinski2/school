\subsubsection*{Dane:}

    n := 9 + 1 = 10 \\
    m := 6 + 1 = 7 \\
    rozpatrujemy trójkąt o bokach (0, 0), (10, 7), (20, 0) \\
    na tym trójkącie zmienna (X, Y) ma stałą gęstość; f(x, y) = C

\subsubsection*{Cel:}

    wyznaczyć gęstość zmiennej T = 2X + Y

\paragraph*{Wyznaczmy C = f(x, y):}
\begin{itemize}
    \item Całka z gęstości po całym obszarze musi wynosić 1
    \item $1 = C * Pole \ trójkąta$ 
    \item $ Pole \ trójkąta = \frac{1}{2} * 20 * 7 = 70$
    \item $C = \frac{1}{70}$
\end{itemize}

\paragraph*{Analiza granic całkowania}
\begin{itemize}
    \item lewy bok trójkąta $ (0, 0) \rightarrow (10, 7) $, równanie prostej opisującej ten bok to $ y = \frac{7}{10}x $, czyli $x = \frac{10}{7} y$ 
    \item prawy bok trójkąt $ (10, 7) \rightarrow (20, 0) $, równanie prostej opisującej ten bok to $ y = -\frac{7}{10}(x - 20) $, czyli $x = 20 - \frac{10}{7}y$
    \item więc nasz obszar całkowania to:
\end{itemize}

\begin{gather*}
0 < y < 7 \\
\frac{10}{7}y  < x < 20 - \frac{10}{7}y
\end{gather*}

\paragraph*{Transformacja} \mbox{}\\
Wprowadźmy nowe zmienne:\\
\\
T = 2X + Y \\ 
S = Y\\
więc:
\begin{itemize}
    \item $X = \frac{T - S}{2}$
    \item $Y = S$
    \item liczymy jakobian tego przekształcenia:
\end{itemize}
$$ J = 
\begin{vmatrix}
    \frac{dx}{dt} & \frac{dx}{ds} \\
    \frac{dy}{dt} & \frac{dy}{ds}
\end{vmatrix} = 
\begin{vmatrix}
    \frac{1}{2} & -\frac{1}{2} \\
    0 & 1
\end{vmatrix}
= \frac{1}{2}
$$
\begin{itemize}
    \item więc nowa gęstość zmiennych T, S wyraża sie wzorem:
\end{itemize}
$$
g(t, s) = f(x(t, s), y(t, s)) * |J| = \frac{1}{70} * \frac{1}{2} = \frac{1}{140}
$$

\paragraph*{Nowe granice całkowania}
\begin{gather*}
0 < S < 7 \\ 
\frac{10}{7}S < \frac{T - S}{2} < 20 - \frac{10}{7}S
\end{gather*}
podzielny ostatnią nierówność i rozwiążmy osobno:
\begin{itemize}
    \item lewa strona:
\end{itemize}
\begin{gather*}
\frac{10}{7}S < \frac{T - S}{2} \\
\frac{20}{7}S < T - S \\ 
S < \frac{7}{27}T
\end{gather*}
\begin{itemize}
    \item prawa strona:
\end{itemize}
\begin{gather*}
\frac{T - S}{2} < 20 - \frac{10}{7}S \\ 
T - S < 40 - \frac{20}{7}S \\ 
S < \frac{7}{13}(40 - T)
\end{gather*}

Więc S jest ograniczone przez: $0 < S < min(7, \frac{7}{27}T, \frac{7}{13}(40 - T))$

Sprawdzmy gdzie te ograniczenia się przecinają:
\begin{gather*}
\frac{7}{27}T = \frac{7}{13}(40 - T) \\
13T = 1080 - 27T \\
T = 27
\end{gather*}

Jak podstawimy T=27 do $\frac{7}{27}T$ to dostajemy 7 więc:
\begin{itemize}
    \item Dla $t \in (0, 27]$ ograniczeniem górnym dla S jest prosta $\frac{7}{27}t$
    \item Dla $t \in (27, 40)$ ograniczeniem górnym dla S jest prosta $\frac{7}{13}(40 - t)$
\end{itemize}

\paragraph*{liczenie gęstości $f_T(t)$}
1. Przypadek $ 0 < t \leq 27 $:
$$
f_T(t) =
\int_0^\frac{7t}{27} \frac{1}{140} \ ds =
\frac{1}{140} [s]_0^\frac{7t}{27} =
\frac{1}{140} * \frac{7}{27}t = \frac{t}{540}
$$

2. Przypadek $ 27 < t < 40 $:
$$
f_T(t) = 
\int_0^{\frac{7}{13}(40 - t)} \frac{1}{140} \ ds = 
\frac{1}{140} [s]_0^{\frac{7}{13}(40 - t)} = 
\frac{1}{140} * \frac{7}{13}(40 - t) = \frac{40 - t}{260}
$$

\subsubsection*{Podsumowanie}
$f_T(t)$ = $\frac{t}{540} $ dla $ t \in (0, 27] $\\
$f_T(t)$ = $\frac{40 - t}{260} $ dla $ t \in (27, 40) $\\
$f_T(t)$ = 0 wpp.
